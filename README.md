# Quick Start

1. Install `manta-style`

```sh
yarn add --dev manta-style
```

2. Create mock api `config.ts`

```ts
interface UserProfile {
  /**
   * @example {{internet.userName}}
   */
  userName: string;
  gender: 0 | 1 | 2;
  /**
   * @example {{date.past}}
   */
  birthday: number;
  /**
   * @example {{address.country}}
   */
  country: string;
  /**
   * @example {{address.state}}
   */
  state: string;
  /**
   * @example {{address.city}}
   */
  city: string;
}
type ApiStatus = 'ok' | 'error';

type WithResponse<T> = {
  status: ApiStatus;
  data: T;
};

export type GET = {
  '/user-profile': WithResponse<UserProfile>;
};
```

3. Start `manta-style`!

```sh
ms -c ./config.ts
```
